# 2026-03-15 GLP-1药物追踪器操作指南

## 一、项目概述与背景分析

### 1.1 GLP-1药物市场背景

GLP-1（胰高血糖素样肽-1）受体激动剂是一类用于治疗2型糖尿病和肥胖症的药物。这类药物包括Ozempic（司美格鲁肽）、Wegovy（司美格鲁肽减肥版）、Mounjaro（替尔泊肽）、Zepbound（替尔泊肽减肥版）等。根据用户原话，有很多人正在使用这些药物进行减肥治疗，市场需求巨大。用户原话："有很多GLP-1应用是由从未真正使用减肥药的技术人员构建的...这个应用是由两位用户（是的，我们都在用药！）在空闲时间构建的。"

### 1.2 项目定位

本项目旨在创建一个真正从用户角度出发、解决真实痛点的GLP-1药物追踪iOS应用。目标用户群体为使用GLP-1类药物（Mounjaro、Zepbound、Wegovy、Ozempic等）进行糖尿病治疗或减肥治疗的人群。根据商机调查报告，目标用户预计500万以上。

**数据模型设计**（可借鉴）：
- **Medication实体**：药物名称、剂量、图标、颜色、每日最大剂量、药效持续时间、间隔时间
- **Dose实体**：服药时间、关联药物ID、状态（进行中/已完成）

```swift
import Foundation
import SwiftData

// 药物模型
@Model
final class Medication {
    var id: UUID
    var name: String                    // 药物名称（如"Mounjaro"）
    var dosage: Double                  // 剂量（如2.5mg）
    var dosageUnit: String              // 剂量单位（如"mg"）
    var frequency: Int                  // 每周注射次数
    var injectionSite: String           // 注射部位（如"腹部"）
    var startDate: Date                 // 开始用药日期
    var notes: String                   // 备注
    var isActive: Bool                  // 是否正在使用
    
    // 关系
    @Relationship(deleteRule: .cascade, inverse: \InjectionRecord.medication)
    var injectionRecords: [InjectionRecord]
    
    init(name: String, dosage: Double, dosageUnit: String, frequency: Int) {
        self.id = UUID()
        self.name = name
        self.dosage = dosage
        self.dosageUnit = dosageUnit
        self.frequency = frequency
        self.injectionSite = "腹部"
        self.startDate = Date()
        self.isActive = true
    }
}

// 注射记录模型
@Model
final class InjectionRecord {
    var id: UUID
    var injectionDate: Date             // 注射日期时间
    var dosage: Double                  // 实际注射剂量
    var site: String                    // 注射部位
    var sideEffects: String             // 副作用记录
    var notes: String                   // 备注
    var mood: String                    // 心情/状态
    var appetiteLevel: Int               // 食欲水平（1-10）
    
    // 关系
    var medication: Medication?
    
    init(injectionDate: Date, dosage: Double, site: String) {
        self.id = UUID()
        self.injectionDate = injectionDate
        self.dosage = dosage
        self.site = site
    }
}
```

```swift
struct AddInjectionSheet: View {
    @Environment(\.modelContext) private var modelContext
    @Environment(\.dismiss) private var dismiss
    
    @Query private var medications: [Medication]
    
    @State private var selectedMedication: Medication?
    @State private var injectionDate = Date()
    @State private var dosage: Double = 2.5
    @State private var injectionSite = "腹部"
    @State private var sideEffects = ""
    @State private var notes = ""
    @State private var appetiteLevel = 5
    
    let injectionSites = ["腹部", "大腿", "上臂", "臀部"]
    
    var body: some View {
        NavigationStack {
            Form {
                // 药物选择
                Section("选择药物") {
                    Picker("药物", selection: $selectedMedication) {
                        Text("选择药物").tag(nil as Medication?)
                        ForEach(medications) { med in
                            Text(med.name).tag(med as Medication?)
                        }
                    }
                }
                
                // 注射详情
                Section("注射详情") {
                    DatePicker("日期时间", selection: $injectionDate)
                    
                    HStack {
                        Text("剂量")
                        TextField("剂量", value: $dosage, format: .number)
                            .keyboardType(.decimalPad)
                        Text(selectedMedication?.dosageUnit ?? "mg")
                    }
                    
                    Picker("注射部位", selection: $injectionSite) {
                        ForEach(injectionSites, id: \.self) { site in
                            Text(site).tag(site)
                        }
                    }
                }
                
                // 副作用记录
                Section("副作用（可选）") {
                    TextField("记录副作用", text: $sideEffects, axis: .vertical)
                        .lineLimit(3)
                    
                    Stepper("食欲水平: \(apppetiteLevel)/10", value: $apppetiteLevel, in: 1...10)
                }
                
                // 备注
                Section("备注") {
                    TextField("添加备注", text: $notes, axis: .vertical)
                        .lineLimit(3)
                }
            }
            .navigationTitle("记录注射")
            .navigationBarTitleDisplayMode(.inline)
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("取消") {
                        dismiss()
                    }
                }
                ToolbarItem(placement: .confirmationAction) {
                    Button("保存") {
                        saveInjection()
                    }
                    .disabled(selectedMedication == nil)
                }
            }
        }
    }
    
    private func saveInjection() {
        guard let medication = selectedMedication else { return }
        
        let record = InjectionRecord(
            injectionDate: injectionDate,
            dosage: dosage,
            site: injectionSite
        )
        record.sideEffects = sideEffects
        record.notes = notes
        record.appetiteLevel = appetiteLevel
        record.medication = medication
        
        modelContext.insert(record)
        
        try? modelContext.save()
        dismiss()
    }
}
```

```swift
import SwiftUI
import SwiftData

struct SideEffectsView: View {
    @Query(sort: \SideEffectRecord.date, order: .reverse) 
    private var sideEffects: [SideEffectRecord]
    
    @State private var showingAddSheet = false
    
    // 按副作用类型分组
    private var groupedSideEffects: [String: [SideEffectRecord]] {
        Dictionary(grouping: sideEffects) { $0.symptom }
    }
    
    var body: some View {
        NavigationStack {
            List {
                // 常见副作用统计
                Section("常见副作用") {
                    ForEach(sideEffectTypes, id: \.self) { type in
                        if let records = groupedSideEffects[type] {
                            SideEffectStatRow(
                                symptom: type,
                                count: records.count,
                                avgSeverity: records.reduce(0) { $0 + $1.severity } / Double(records.count)
                            )
                        }
                    }
                }
                
                // 详细记录
                Section("所有记录") {
                    ForEach(sideEffects) { record in
                        SideEffectRowView(record: record)
                    }
                }
            }
            .navigationTitle("副作用追踪")
            .toolbar {
                Button {
                    showingAddSheet = true
                } label: {
                    Image(systemName: "plus")
                }
            }
            .sheet(isPresented: $showingAddSheet) {
                AddSideEffectSheet()
            }
        }
    }
    
    // GLP-1常见副作用类型
    private var sideEffectTypes: [String] {
        ["恶心", "呕吐", "腹泻", "便秘", "腹痛", "食欲减退", "疲劳", "头痛", "头晕", "胃灼热"]
    }
}

struct SideEffectStatRow: View {
    let symptom: String
    let count: Int
    let avgSeverity: Double
    
    var body: some View {
        HStack {
            VStack(alignment: .leading) {
                Text(symptom)
                    .font(.headline)
                Text("发生 \(count) 次")
                    .font(.caption)
                    .foregroundColor(.secondary)
            }
            
            Spacer()
            
            // 严重程度指示器
            HStack(spacing: 4) {
                ForEach(1...10, id: \.self) { level in
                    Circle()
                        .fill(level <= Int(avgSeverity) ? severityColor : Color.gray.opacity(0.3))
                        .frame(width: 8, height: 8)
                }
            }
        }
    }
    
    private var severityColor: Color {
        if avgSeverity <= 3 { return .green }
        if avgSeverity <= 6 { return .yellow }
        return .red
    }
}
```

#### 4.3.4 HealthKit集成

```swift
import HealthKit
import SwiftUI

class HealthKitManager: ObservableObject {
    let healthStore = HKHealthStore()
    
    // 检查HealthKit是否可用
    var isHealthKitAvailable: Bool {
        HKHealthStore.isHealthDataAvailable()
    }
    
    // 请求授权
    func requestAuthorization() async throws {
        guard isHealthKitAvailable else {
            throw HealthKitError.notAvailable
        }
        
        let typesToRead: Set<HKObjectType> = [
            HKObjectType.quantityType(forIdentifier: .bodyMass)!,
            HKObjectType.quantityType(forIdentifier: .stepCount)!,
            HKObjectType.quantityType(forIdentifier: .activeEnergyBurned)!
        ]
        
        try await healthStore.requestAuthorization(toShare: [], read: typesToRead)
    }
    
    // 从HealthKit读取体重数据
    func fetchWeightData(from startDate: Date) async throws -> [Double] {
        guard let weightType = HKObjectType.quantityType(forIdentifier: .bodyMass) else {
            throw HealthKitError.typeNotAvailable
        }
        
        let predicate = HKQuery.predicateForSamples(
            withStart: startDate,
            end: Date(),
            options: .strictStartDate
        )
        
        return try await withCheckedThrowingContinuation { continuation in
            let query = HKSampleQuery(
                sampleType: weightType,
                predicate: predicate,
                limit: HKObjectQueryNoLimit,
                sortDescriptors: [NSSortDescriptor(key: HKSampleSortIdentifierStartDate, ascending: true)]
            ) { _, samples, error in
                if let error = error {
                    continuation.resume(throwing: error)
                    return
                }
                
                guard let samples = samples as? [HKQuantitySample] else {
                    continuation.resume(returning: [])
                    return
                }
                
                let weights = samples.map { sample in
                    sample.quantity.doubleValue(for: .gramUnit(with: .kilo))
                }
                continuation.resume(returning: weights)
            }
            
            healthStore.execute(query)
        }
    }
    
    // 同步体重到HealthKit
    func saveWeight(weight: Double, date: Date) async throws {
        guard let weightType = HKObjectType.quantityType(forIdentifier: .bodyMass) else {
            throw HealthKitError.typeNotAvailable
        }
        
        let quantity = HKQuantity(unit: .gramUnit(with: .kilo), doubleValue: weight)
        let sample = HKQuantitySample(
            type: weightType,
            quantity: quantity,
            start: date,
            end: date
        )
        
        try await healthStore.save(sample)
    }
}

enum HealthKitError: Error {
    case notAvailable
    case typeNotAvailable
    case authorizationDenied
}
```

```swift
import UserNotifications
import SwiftUI

class NotificationManager: ObservableObject {
    @Published var isAuthorized = false
    
    init() {
        Task {
            await checkAuthorization()
        }
    }
    
    // 检查授权状态
    func checkAuthorization() async {
        let settings = await UNUserNotificationCenter.current().notificationSettings()
        await MainActor.run {
            isAuthorized = settings.authorizationStatus == .authorized
        }
    }
    
    // 请求授权
    func requestAuthorization() async throws {
        try await UNUserNotificationCenter.current().requestAuthorization(
            options: [.alert, .badge, .sound]
        )
        await checkAuthorization()
    }
    
    // 创建注射提醒
    func scheduleInjectionReminder(
        medicationName: String,
        nextInjectionDate: Date,
        dosage: String,
        identifier: String
    ) {
        let content = UNMutableNotificationContent()
        content.title = "💉 该注射了"
        content.body = "\(medicationName) - \(dosage)"
        content.sound = .default
        content.badge = 1
        
        // 提前15分钟提醒
        let triggerDate = nextInjectionDate.addingTimeInterval(-15 * 60)
        
        // 只在将来时间设置提醒
        guard triggerDate > Date() else { return }
        
        let components = Calendar.current.dateComponents(
            [.year, .month, .day, .hour, .minute],
            from: triggerDate
        )
        
        let trigger = UNCalendarNotificationTrigger(dateMatching: components, repeats: false)
        
        let request = UNNotificationRequest(
            identifier: identifier,
            content: content,
            trigger: trigger
        )
        
        UNUserNotificationCenter.current().add(request) { error in
            if let error = error {
                print("提醒创建失败: \(error)")
            }
        }
    }
    
    // 创建每周提醒（用于记录体重）
    func scheduleWeeklyWeightReminder(weekday: Int = 1) {  // 默认周一
        let content = UNMutableNotificationContent()
        content.title = "📝 记录体重"
        content.body = "别忘了记录今天的体重！"
        content.sound = .default
        
        var components = DateComponents()
        components.weekday = weekday
        components.hour = 9
        components.minute = 0
        
        let trigger = UNCalendarNotificationTrigger(dateMatching: components, repeats: true)
        
        let request = UNNotificationRequest(
            identifier: "weekly-weight-reminder",
            content: content,
            trigger: trigger
        )
        
        UNUserNotificationCenter.current().add(request)
    }
    
    // 取消所有提醒
    func cancelAllReminders() {
        UNUserNotificationCenter.current().removeAllPendingNotificationRequests()
    }
    
    // 取消特定提醒
    func cancelReminder(identifier: String) {
        UNUserNotificationCenter.current().removePendingNotificationRequests(withIdentifiers: [identifier])
    }
}
```

## 附录：常见问题解答

### Q1: 是否需要后端服务器？
A: 不需要。本项目采用完全本地存储方案，使用SwiftData进行数据持久化。如需跨设备同步，可选配CloudKit。

### Q2: 如何处理HealthKit权限？
A: 需要在Info.plist中配置NSHealthShareUsageDescription和NSHealthUpdateUsageDescription描述，并在应用内引导用户授权。

### Q3: App Store审核有哪些注意事项？
A: 健康类应用需注意：1）提供隐私政策；2）明确说明不提供医疗建议；3）避免声称能治疗或预防疾病。

### Q4: 如何处理数据导出？
A: 提供CSV导出功能和PDF医生报告生成功能，数据格式需符合医疗记录标准。
